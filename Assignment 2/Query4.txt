select S.S_NAME as name, count(DISTINCT C.C_CUSTKEY)
from SUPPLIER S, ORDERS O, LINEITEM L, CUSTOMER C,PARTSUPP P
where (S.S_SUPPKEY == P.PS_SUPPKEY and P.PS_SUPPKEY == L.L_SUPPKEY and O.O_ORDERKEY == L.L_ORDERKEY
    and C.C_CUSTKEY == O.O_CUSTKEY) group by S.S_SUPPKEY having count(DISTINCT C.C_CUSTKEY) > 615
ORDER BY count(DISTINCT C.C_CUSTKEY) DESC
